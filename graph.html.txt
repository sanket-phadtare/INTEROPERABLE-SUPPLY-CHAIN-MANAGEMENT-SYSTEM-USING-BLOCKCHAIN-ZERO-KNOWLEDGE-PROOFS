  <div class="fig-title">Fig. 1. Verification Time vs Number of Verification Requests</div>
  <div class="chart-area">
    <svg id="chart" viewBox="0 0 780 480" xmlns="http://www.w3.org/2000/svg"></svg>
  </div>
</div>

<script>
const svg = document.getElementById('chart');
const NS = 'http://www.w3.org/2000/svg';

// Layout
const margin = { top: 24, right: 160, bottom: 68, left: 80 };
const W = 780, H = 480;
const plotW = W - margin.left - margin.right;
const plotH = H - margin.top - margin.bottom;

// Data
const traditional = [
  {x: 10,  y: 4},
  {x: 50,  y: 20},
  {x: 100, y: 41},
  {x: 200, y: 82},
  {x: 500, y: 205},
];
const zkp = [
  {x: 10,  y: 0.2},
  {x: 50,  y: 1},
  {x: 100, y: 2},
  {x: 200, y: 4},
  {x: 500, y: 10},
];

// Scales
const xMin = 0, xMax = 500;
const yMin = 0, yMax = 220;
const xTicks = [0, 100, 200, 300, 400, 500];
const yTicks = [0, 20, 40, 60, 80, 100, 120, 140, 160, 180, 200, 220];

function scaleX(v) { return margin.left + (v - xMin) / (xMax - xMin) * plotW; }
function scaleY(v) { return margin.top + plotH - (v - yMin) / (yMax - yMin) * plotH; }

function el(tag, attrs={}, parent=svg) {
  const e = document.createElementNS(NS, tag);
  for (const [k, v] of Object.entries(attrs)) e.setAttribute(k, v);
  parent.appendChild(e);
  return e;
}

// White background
el('rect', {x:0, y:0, width:W, height:H, fill:'#ffffff'});

// Clip path
const defs = el('defs');
const clip = el('clipPath', {id:'plot-clip'}, defs);
el('rect', {x: margin.left, y: margin.top, width: plotW, height: plotH}, clip);

// Plot background
el('rect', {
  x: margin.left, y: margin.top,
  width: plotW, height: plotH,
  fill: '#ffffff',
  stroke: '#000000',
  'stroke-width': '1.2'
});

// Gridlines
for (const t of xTicks) {
  if (t === xMin) continue;
  const x = scaleX(t);
  el('line', {
    x1: x, y1: margin.top,
    x2: x, y2: margin.top + plotH,
    stroke: '#cccccc', 'stroke-width': '0.7',
    'stroke-dasharray': '3,3'
  });
}
for (const t of yTicks) {
  if (t === yMin) continue;
  const y = scaleY(t);
  el('line', {
    x1: margin.left, y1: y,
    x2: margin.left + plotW, y2: y,
    stroke: '#cccccc', 'stroke-width': '0.7',
    'stroke-dasharray': '3,3'
  });
}

// Axes
// X axis
el('line', {
  x1: margin.left, y1: margin.top + plotH,
  x2: margin.left + plotW, y2: margin.top + plotH,
  stroke: '#000', 'stroke-width': '1.4'
});
// Y axis
el('line', {
  x1: margin.left, y1: margin.top,
  x2: margin.left, y2: margin.top + plotH,
  stroke: '#000', 'stroke-width': '1.4'
});

// Tick labels — X
for (const t of xTicks) {
  const x = scaleX(t);
  const y = margin.top + plotH;
  el('line', {x1: x, y1: y, x2: x, y2: y+5, stroke:'#000','stroke-width':'1'});
  el('text', {
    x, y: y+18,
    'text-anchor':'middle',
    'font-size':'12',
    'font-family':'Times New Roman, Times, serif',
    fill:'#000'
  }).textContent = t;
}

// Tick labels — Y
for (const t of yTicks) {
  const x = margin.left;
  const y = scaleY(t);
  el('line', {x1: x-5, y1: y, x2: x, y2: y, stroke:'#000','stroke-width':'1'});
  el('text', {
    x: x-10,
    y: y+4,
    'text-anchor':'end',
    'font-size':'12',
    'font-family':'Times New Roman, Times, serif',
    fill:'#000'
  }).textContent = t;
}

// Axis labels
// X
el('text', {
  x: margin.left + plotW/2,
  y: H - 10,
  'text-anchor':'middle',
  'font-size':'13.5',
  'font-family':'Times New Roman, Times, serif',
  'font-weight':'bold',
  fill:'#000'
}).textContent = 'Number of Verification Requests';

// Y (rotated)
const yLabel = el('text', {
  x: 0, y: 0,
  'text-anchor':'middle',
  'font-size':'13.5',
  'font-family':'Times New Roman, Times, serif',
  'font-weight':'bold',
  fill:'#000',
  transform: `translate(16, ${margin.top + plotH/2}) rotate(-90)`
});
yLabel.textContent = 'Total Verification Time (seconds)';

// Helper: build polyline path
function pointsStr(data) {
  return data.map(d => `${scaleX(d.x)},${scaleY(d.y)}`).join(' ');
}

// Lines
el('polyline', {
  points: pointsStr(traditional),
  fill:'none',
  stroke:'#1a5eb8',
  'stroke-width':'2.2',
  'stroke-linejoin':'round',
  'stroke-linecap':'round',
  'clip-path':'url(#plot-clip)'
});

el('polyline', {
  points: pointsStr(zkp),
  fill:'none',
  stroke:'#1a9641',
  'stroke-width':'2.2',
  'stroke-linejoin':'round',
  'stroke-linecap':'round',
  'clip-path':'url(#plot-clip)'
});

// Markers
function drawMarkers(data, color) {
  for (const d of data) {
    el('circle', {
      cx: scaleX(d.x), cy: scaleY(d.y),
      r: '5',
      fill: '#ffffff',
      stroke: color,
      'stroke-width': '2',
      'clip-path': 'url(#plot-clip)'
    });
  }
}
drawMarkers(traditional, '#1a5eb8');
drawMarkers(zkp, '#1a9641');

// Legend — top right inside plot
const lx = margin.left + plotW + 12;
const ly = margin.top + 16;
const lw = 140, lh = 68;

el('rect', {
  x: lx, y: ly,
  width: lw, height: lh,
  fill:'#ffffff',
  stroke:'#000000',
  'stroke-width':'0.9',
  rx:'2', ry:'2'
});

// Traditional legend
el('line', {
  x1: lx+12, y1: ly+22,
  x2: lx+44, y2: ly+22,
  stroke:'#1a5eb8','stroke-width':'2.2'
});
el('circle', {
  cx: lx+28, cy: ly+22,
  r:'4.5',
  fill:'#ffffff',
  stroke:'#1a5eb8','stroke-width':'2'
});
el('text', {
  x: lx+52, y: ly+26,
  'font-size':'11.5',
  'font-family':'Times New Roman, Times, serif',
  fill:'#000'
}).textContent = 'Traditional';

// ZKP legend
el('line', {
  x1: lx+12, y1: ly+48,
  x2: lx+44, y2: ly+48,
  stroke:'#1a9641','stroke-width':'2.2'
});
el('circle', {
  cx: lx+28, cy: ly+48,
  r:'4.5',
  fill:'#ffffff',
  stroke:'#1a9641','stroke-width':'2'
});
el('text', {
  x: lx+52, y: ly+52,
  'font-size':'11.5',
  'font-family':'Times New Roman, Times, serif',
  fill:'#000'
}).textContent = 'ZKP-Based';

// Border around entire figure
el('rect', {
  x: 1, y: 1,
  width: W-2, height: H-2,
  fill:'none',
  stroke:'#ffffff',
  'stroke-width':'2'
});
</script>
